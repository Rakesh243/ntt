# app.nim
import jester, karax [*, VNode], logging, times, strformat, db_sqlite

var db: DbConn

# Initialize the database and create a users table if not exists
proc initDatabase() =
  db = open("typing_test.db")
  exec(db, """
    CREATE TABLE IF NOT EXISTS users (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      username TEXT NOT NULL,
      speed INTEGER NOT NULL,
      time TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    );
  """)

# Insert user data after test
proc insertUserData(username: string, speed: int) =
  let query = "INSERT INTO users (username, speed) VALUES (?, ?);"
  exec(db, query, username, $speed)

# Serve the frontend (Karax HTML) to the client
routes:
  get "/":
    resp h2("Typing Speed Test")
    resp """
    <div id="karax-app"></div>
    <script type="text/javascript" src="/static/main.js"></script>
    """
  
  post "/submit":
    let username = paramStr("username")
    let speed = paramStr("speed").parseInt
    insertUserData(username, speed)
    resp "Data received. User: " & username & ", Speed: " & $speed & " WPM."

  get "/scores":
    let query = "SELECT username, speed, time FROM users ORDER BY speed DESC LIMIT 10;"
    let rows = db.getRows(query)
    resp "<h3>Top 10 Scores</h3>"
    resp "<ul>"
    for row in rows:
      resp f"<li>{row["username"]}: {row["speed"]} WPM at {row["time"]}</li>"
    resp "</ul>"

# Serve static files (JavaScript generated by Karax)
serveStaticDir "/static", "public"

proc main() =
  initDatabase()
  let port = 8080
  logInfo(fmt"Server running on port {port}")
  runForever()

# Words to test typing speed on
let sampleText = "The quick brown fox jumps over the lazy dog"

proc startTest() =
  startTime = epochTime()

  typedWords = @[]

proc calculateSpeed(): int =
  let endTime = epochTime()
  let timeTaken = (endTime -startTime).float / 60.0 # Minutes
  return int(typedWords.len.float / timeTaken) # Words per minute
